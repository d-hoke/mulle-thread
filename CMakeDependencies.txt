if( NOT __MULLE_THREAD_CMAKE_DEPENDENCIES_TXT__)
   set( __MULLE_THREAD_CMAKE_DEPENDENCIES_TXT__ ON)

   message( STATUS "# Include mulle-thread CMakeDependencies.txt")

   if( NOT MULLE_THREAD_LIBRARY)
      find_library( MULLE_THREAD_LIBRARY NAMES mulle_thread)
      message(STATUS "MULLE_THREAD_LIBRARY is ${MULLE_THREAD_LIBRARY}")
      set( C_DEPENDENCY_LIBRARIES
         ${MULLE_THREAD_LIBRARY}
         ${C_DEPENDENCY_LIBRARIES}
         CACHE INTERNAL "need to cache this"
      )
   endif()

   if( UNIX)
      #
      # Pthread is OS specific and not a C dependency,
      # because it's not bootstrapped
      #
      if( NOT PTHREAD_LIBRARY)
         find_library( PTHREAD_LIBRARY NAMES pthread pthreads)
         message(STATUS "PTHREAD_LIBRARY is ${PTHREAD_LIBRARY}")
         set( OS_SPECIFIC_LIBRARIES
            ${PTHREAD_LIBRARY}
            ${OS_SPECIFIC_LIBRARIES}
            CACHE INTERNAL "need to cache this"
         )
      endif()
   endif()
endif()
